Softwareentwicklung 1

Die Aufgabe

  • Wir wollen den Umfang und die Fläche von Rechtecken berechnen. Als Input verwenden wir Länge und Breite mit Kommazahlen als Werte.
  • Die Ergebnisse sollen folgendermaßen mit println() angezeigt werden: 
    Rechteck mit Länge = 100.0 und Breite = 50.0:
Der Umfang beträgt: 300.0
Die Fläche beträgt: 5000.0
  • Weiters sollen die Rechtecke in einer zufälligen Farbe untereinander gezeichnet werden. Der Abstand zwischen den Rechtecken soll 15 Punkte sein.
  • Erstelle für jede Berechnung je eine eigene Funktion (berechneUmfang(...) & berechneFlaeche(...)), die den berechneten Wert zurückgibt.
  • Erstelle weiters eine Funktion: berechneRechteck(...), die die oberen 2 Funktionen verwendet, die Ergebnisse anzeigt und das Rechteck zeichnet.
  • In setup() soll die Funktion berechneRechteck() 3 x mit verschiedenen Werten aufgerufen werden.

Die Lösung


/*
  Codebeispiel für Berechnung von Umfang & Fläche von Rechtecken
 */

// --- Setup und Hauptlogik (wird einmal zu Beginn ausgeführt) ---
void setup() {
  size(400, 400); // Zeichenfläche groß genug für unsere 3 Rechtecke
  background(240); // Helle Hintergrundfarbe

  // Berechnung und Anzeige von 3 Rechtecken mit der Haupt-Funktion.
  // Die "Breite" ist beim Zeichnen die Höhe.
  berechneRechteck(100.0, 50.0, 0);  // Argumente für breite, laenge, yPos
  // Der Wert für yPos soll unter dem vorigen Rechteck + 15px Abstand sein
  berechneRechteck(200.0, 150.5, (50 + 15));
  berechneRechteck(30.753, 73.53, (50 + 15 + 151 + 15));
}

/*
 Funktion zur Berechnung des Umfangs eines Rechtecks.
 Die Funktion hat 2 float Parameter.
 Das Ergebnis wird als Kommazahl (float) zurückgegeben.
 */
float berechneUmfang(float laenge, float breite) {
  // Die Formel: Umfang = 2 * (Länge + Breite)
  float umfang = 2 * (laenge + breite); // Ergebnis zwischenspeichern
  return umfang; // Rückgabe des Ergebnisses
}

/*
 Funktion zur Berechnung der Fläche eines Rechtecks.
 Die Funktion hat 2 float Parameter.
 Das Ergebnis wird als Kommazahl (float) zurückgegeben.
 */
float berechneFlaeche(float laenge, float breite) {
  // Die Formel: Fläche = Länge * Breite
  float flaeche = laenge * breite; // Ergebnis zwischenspeichern
  return flaeche; // Rückgabe des Ergebnisses
}

// Haupt-Funktion für das Berechnen, Anzeigen und Zeichnen der Rechtecke
// Diese Funktion hat keine Rückgabe, somit void.
void berechneRechteck(float laenge, float breite, int yPos) {
  // 1. Wir definieren die Abmessungen unseres Rechtecks (mit Dezimalstellen)

  // 2. Wir rufen die Funktionn mit den Abmessungen auf
  float meinUmfang = berechneUmfang(laenge, breite);
  float meineFlaeche = berechneFlaeche(laenge, breite);

  // 3. Wir geben die Ergebnisse in der Konsole aus
  println();  // Wir möchten eine Leerzeile :)!
  println("Rechteck mit Länge = " + laenge + " und Breite = " + breite + ":");
  println("Der Umfang beträgt: " + meinUmfang);
  println("Die Fläche beträgt: " + meineFlaeche);

  // 4. Wir zeigen die Ergebnisse auf dem Processing-Canvas an
  int randColor = color(
    randomZahl(100, 255),
    randomZahl(100, 255),
    randomZahl(100, 255)
    );
  fill(randColor);
  rect(10, 10 + yPos, laenge, breite);
}

/*
 Eine Hilfs-Funktion,
 die das Erzeugen von Random-Zahlen mit min & max erleichtert.
 */
int randomZahl(int min, int max) {
  // Formal kopiert von den Folien!
  int rand = min + (int)(Math.random() * ((max - min) + 1));
  return rand;
}
